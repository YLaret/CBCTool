CREATE TABLE TARGET (
    TargetID INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL
);

CREATE TABLE EIGENSCHAP (
    EigID INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL
);

CREATE TABLE HEEFT (
    HeeftID INTEGER PRIMARY KEY AUTOINCREMENT,
    TargetID INTEGER NOT NULL,
    EigID INTEGER NOT NULL,
    Weight REAL NOT NULL,
    Min REAL NOT NULL,
    Max REAL NOT NULL,
    FOREIGN KEY (TargetID) REFERENCES TARGET (TargetID),
    FOREIGN KEY (EigID) REFERENCES EIGENSCHAP (EigID)
);
